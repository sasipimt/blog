searches = 'matterhorn','assiniboine'
path = Path('matterhorn_or_not')

for o in searches:

    download_images(dest, urls=search_images(f'{o} in winter'))
    time.sleep(5)
    resize_images(path/o, max_size=400, dest=path/o)
